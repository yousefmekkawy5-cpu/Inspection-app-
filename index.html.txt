<script src="https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js"></script>

<script>
let inspections = JSON.parse(localStorage.getItem("inspections")) || [];

function calculateScore(data){
  let total = (+data.walls + +data.electric + +data.plumbing);
  return Math.round((total/15)*100);
}

function saveInspection(){
  const data = {
    client: client.value,
    location: location.value,
    type: type.value,
    walls: walls.value,
    electric: electric.value,
    plumbing: plumbing.value,
    notes: notes.value,
    score: 0,
    date: new Date().toLocaleString()
  };

  data.score = calculateScore(data);
  inspections.push(data);
  localStorage.setItem("inspections", JSON.stringify(inspections));

  alert("تم حفظ التقرير ✅");
}

async function generatePDF(){
  const { jsPDF } = window.jspdf;
  const doc = new jsPDF();

  let last = inspections[inspections.length -1];

  doc.text("Property Inspection Report", 10, 10);
  doc.text("Client: " + last.client, 10, 20);
  doc.text("Location: " + last.location, 10, 30);
  doc.text("Score: " + last.score + "%", 10, 40);
  doc.text("Notes: " + last.notes, 10, 50);

  doc.save("report.pdf");
}

if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('service-worker.js');
}
